## Changing Value

<Preview>
  <Story name="Changing Value">
    {
      () => {
        const reducer = val => val + 1
        const [value, update] = useReducer(reducer, 0)
        useEffect(() => {
          setInterval(update, 1000)
        }, [])
        return <Subject label="changing value" value={value.toString()} />
      }
    }
  </Story>
