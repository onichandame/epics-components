# Thumb Wheeler

A simple thumb wheeler used to fine adjust values.

## Basic

<Preview>
  <Story name="Normal">
      {
        () => {
          const reduce = (state, action) => {
            switch(action) {
              case 'increment':
                return state + 1
              case 'decrement':
                return state - 1
              default:
                throw new Error(`unknown action ${action}`)
            }
          }
          const [value, dispatch] = useReducer(reduce, 0)
          return (
            <Subject
              value={value}
              increment={() => dispatch('increment')}
              decrement={() => dispatch('decrement')}
            />
          )
        }
      }
  </Story>
</Preview>
